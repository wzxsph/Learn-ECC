## utils.js

**เส้นทาง**: `scripts/lib/utils.js`

ไลบรารีฟังก์ชันข้ามแพลตฟอร์ม, ใช้งานได้บน Windows, macOS และ Linux

### การตรวจจับแพลตฟอร์ม

```javascript
const { isWindows, isMacOS, isLinux } = require('./utils');

// isWindows: true บน Windows
// isMacOS: true บน macOS
// isLinux: true บน Linux
```

### การดำเนินการไดเรกทอรี

```javascript
const {
  getHomeDir,      // รับไดเรกทอรีหลักของผู้ใช้
  getClaudeDir,    // รับไดเรกทอรี ~/.claude
  getSessionsDir,  // รับไดเรกทอรี ~/.claude/session-data
  getTempDir,      // รับไดเรกทอรีชั่วคราว
  ensureDir,       // ตรวจสอบว่าไดเรกทอรีมีอยู่, สร้างถ้าไม่มี
  findFiles        // ค้นหาไฟล์ที่ตรงรูปแบบ
} = require('./utils');

// สร้างไดเรกทอรี (ถ้าไม่มี)
ensureDir('/path/to/dir');

// ค้นหาไฟล์
const files = findFiles('/path', '*.js', { recursive: true });
```

### วันที่และเวลา

```javascript
const {
  getDateString,      // ส่งคืนรูปแบบ YYYY-MM-DD
  getTimeString,      // ส่งคืนรูปแบบ HH:MM
  getDateTimeString   // ส่งคืนรูปแบบ YYYY-MM-DD HH:MM:SS
} = require('./utils');
```

### เซสชันและโครงการ

```javascript
const {
  sanitizeSessionId,    // ทำความสะอาด session ID, ลบอักขระที่ไม่ถูกต้อง
  getSessionIdShort,    // รับ session ID สั้น
  getGitRepoName,       // รับชื่อ git repository
  getProjectName        // รับชื่อโครงการ
} = require('./utils');

// ทำความสะอาด session ID
const cleanId = sanitizeSessionId('my-session@123'); // 'my-session-123'

// รับชื่อโครงการ
const project = getProjectName();
```

### การดำเนินการไฟล์

```javascript
const {
  readFile,        // อ่านไฟล์อย่างปลอดภัย, ล้มเหลวส่งคืน null
  writeFile,       // เขียนไฟล์
  appendFile,      // ต่อท้ายไฟล์
  replaceInFile,   // แทนที่เนื้อหาไฟล์
  countInFile,     // นับจำนวนรูปแบบที่ปรากฏ
  grepFile         // ค้นหารูปแบบในไฟล์
} = require('./utils');

// อ่านไฟล์
const content = readFile('/path/to/file.txt');

// แทนที่เนื้อหาไฟล์ (การจับคู่ครั้งแรก)
replaceInFile('/path', 'old', 'new');

// แทนที่การจับคู่ทั้งหมด
replaceInFile('/path', 'old', 'new', { all: true });

// นับจำนวนครั้งที่ปรากฏ
const count = countInFile('/path', /pattern/g);

// ค้นหาและส่งคืนบรรทัดที่ตรง
const matches = grepFile('/path', /regex/);
// ส่งคืน: [{ lineNumber: 1, content: '...' }]
```

### การประมวลผลสตริงง

```javascript
const { stripAnsi } = require('./utils');

// ลบลำดับ ANSI escape (สี ฯลฯ)
const clean = stripAnsi('\x1b[32mHello\x1b[0m'); // 'Hello'
```

### Hook I/O

```javascript
const { readStdinJson, log, output } = require('./utils');

// อ่าน JSON จาก stdin (สำหรับ hooks)
const input = await readStdinJson({ timeoutMs: 5000 });

// ส่งออกไปยัง stderr (มองเห็นโดยผู้ใช้)
log('Debug message');

// ส่งออกไปยัง stdout (ส่งคืนให้ Claude)
output({ result: 'data' });
```

### คำสั่งระบบ

```javascript
const { commandExists, runCommand, isGitRepo, getGitModifiedFiles } = require('./utils');

// ตรวจสอบว่าคำสั่งมีอยู่
if (commandExists('git')) { /* ... */ }

// รันคำสั่ง (เฉพาะคำสั่ง whitelist)
const result = runCommand('git rev-parse --show-toplevel');
if (result.success) {
  console.log(result.output);
}

// ตรวจสอบว่าเป็น git repository
if (isGitRepo()) { /* ... */ }

// รับไฟล์ที่แก้ไข
const files = getGitModifiedFiles(['\\.js$', '\\.md$']);
```

## package-manager.js

**เส้นทาง**: `scripts/lib/package-manager.js`

เครื่องมือตรวจจับและเลือก package manager

### Package managers ที่รองรับ

- npm
- pnpm
- yarn
- bun

### ลำดับความสำคัญในการตรวจจับ

1. ตัวแปรสิ่งแวดล้อม `CLAUDE_PACKAGE_MANAGER`
2. การตั้งค่าโปรเจกต์ `.claude/package-manager.json`
3. ฟิลด์ `packageManager` ของ `package.json`
4. การตรวจจับจาก lock file
5. ค่ากำหนดของผู้ใช้ทั่วโลก `~/.claude/package-manager.json`
6. ค่าเริ่มต้น npm

### API หลัก

```javascript
const {
  getPackageManager,           // รับ package manager ของโครงการปัจจุบัน
  setPreferredPackageManager,  // ตั้งค่าค่ากำหนดทั่วโลก
  setProjectPackageManager,    // ตั้งค่าค่ากำหนดโปรเจกต์
  getAvailablePackageManagers, // รับ package managers ที่ใช้ได้บนระบบ
  detectFromLockFile,          // ตรวจจับจาก lock file
  detectFromPackageJson,       // ตรวจจับจาก package.json
  getRunCommand,               // รับคำสั่งสำหรับรันสคริปต์
  getExecCommand,              // รับคำสั่งสำหรับรัน binary
  getSelectionPrompt,          // รับ prompt การเลือกแบบโต้ตอบ
  getCommandPattern            // สร้าง regular expression สำหรับจับคู่คำสั่ง
} = require('./package-manager');

// รับ package manager ปัจจุบัน
const { name, config, source } = getPackageManager();
// name: 'pnpm', 'npm', 'yarn', หรือ 'bun'
// config: { installCmd, runCmd, execCmd, testCmd, ... }
// source: 'environment', 'project-config', 'lock-file', ...

// ตั้งค่าค่ากำหนดทั่วโลก
setPreferredPackageManager('pnpm');

// ตั้งค่าค่ากำหนดโปรเจกต์
setProjectPackageManager('yarn', '/path/to/project');

// รับคำสั่งสำหรับรัน
const testCmd = getRunCommand('test'); // 'pnpm test' หรือ 'npm test' ฯลฯ
const buildCmd = getRunCommand('build');

// รับคำสั่งสำหรับรัน binary
const eslintCmd = getExecCommand('eslint', '--fix .');
```

### การตรวจสอบความปลอดภัย

`getRunCommand` และ `getExecCommand` ตรวจสอบข้อมูลนำเข้า:
- ชื่อสคริปต์อนุญาตเฉพาะตัวอักษร ตัวเลข dash underscore dot slash และ @
- พารามิเตอร์อนุญาตเฉพาะตัวอักษร ตัวเลข ช่องว่าง dash dots slashes equals colons commas quotes และ @

## session-manager.js

**เส้นทาง**: `scripts/lib/session-manager.js`

ไลบรารีจัดการเซสชัน, ให้ CRUD operations สำหรับเซสชัน

### การจัดเก็บเซสชัน

- รูปแบบใหม่: `~/.claude/session-data/YYYY-MM-DD-[session-id]-session.tmp`
- ความเข้ากันได้ของรูปแบบเก่า: `~/.claude/sessions/YYYY-MM-DD-session.tmp`

### API หลัก

```javascript
const {
  parseSessionFilename,      // แยกวิเคราะห์ชื่อไฟล์เซสชัน
  getSessionPath,            // รับ path เต็มของไฟล์เซสชัน
  getSessionContent,         // อ่านเนื้อหาเซสชัน
  parseSessionMetadata,      // แยกวิเคราะห์ metadata ของเซสชัน
  getSessionStats,           // รับสถิติเซสชัน
  getSessionTitle,           // รับชื่อเซสชัน
  getSessionSize,            // รับขนาดเซสชัน (รูปแบบที่มนุษย์อ่านได้)
  getAllSessions,            // รับเซสชันทั้งหมด (รองรับการแบ่งหน้าและกรอง)
  getSessionById,            // รับเซสชันเดียวตาม ID
  writeSessionContent,       // เขียนเนื้อหาเซสชัน
  appendSessionContent,      // ต่อท้ายเนื้อหาเซสชัน
  deleteSession,             // ลบเซสชัน
  sessionExists              // ตรวจสอบว่าเซสชันมีอยู่
} = require('./session-manager');

// รับเซสชันทั้งหมด
const { sessions, total, offset, limit, hasMore } = getAllSessions({
  limit: 50,
  offset: 0,
  date: '2026-05-26',
  search: 'abc123'
});

// รับเซสชันเดียว
const session = getSessionById('abc123', true);
// session: { filename, shortId, date, sessionPath, content, metadata, stats, ... }

// แยกวิเคราะห์ metadata ของเซสชัน
const metadata = parseSessionMetadata(content);
// metadata: { title, date, started, lastUpdated, project, branch, worktree, completed, inProgress, notes, context }

// รับสถิติเซสชัน
const stats = getSessionStats(sessionPathOrContent);
// stats: { totalItems, completedItems, inProgressItems, lineCount, hasNotes, hasContext }

// เขียนเซสชัน
writeSessionContent(sessionPath, '# New Session\n\nContent...');

// ต่อท้ายเนื้อหา
appendSessionContent(sessionPath, '\n\nMore content...');

// ลบเซสชัน
deleteSession(sessionPath);
```

### รูปแบบ Metadata ของเซสชัน

Metadata ในเนื้อหาเซสชัน:

```markdown
# Session Title

**Date:** 2026-05-26
**Started:** 14:30
**Last Updated:** 15:45
**Project:** my-project
**Branch:** main
**Worktree:** frontend

### Completed
- [x] Task 1
- [x] Task 2

### In Progress
- [ ] Task 3

### Notes for Next Session
Remember to check the config file.

### Context to Load
```
previous-context-content
```
```
