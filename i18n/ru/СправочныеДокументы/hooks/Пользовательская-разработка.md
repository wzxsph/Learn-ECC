### HookInput 

```typescript
interface HookInput {
  tool_name: string;          // Название
  tool_input: {               // 
    command?: string;         // Bash: 
    file_path?: string;        // Edit/Write/Read: 
    old_string?: string;       // Edit: 
    new_string?: string;       // Edit: 
    content?: string;          // Write: 
  };
  tool_output?: {             // PostToolUse only
    output?: string;          // /
  };
}
```

### 

```javascript
process.stdin.on('end', () => {
  const input = JSON.parse(data);

  // Название
  console.log('Tool:', input.tool_name);

  // Тип
  if (input.tool_name === 'Bash') {
    console.log('Command:', input.tool_input.command);
  }

  if (input.tool_name === 'Edit' || input.tool_name === 'Write') {
    console.log('File:', input.tool_input.file_path);
  }

  // PostToolUse
  if (input.tool_output) {
    console.log('Output:', input.tool_output.output);
  }
});
```
