## package-manager.js

****: `scripts/lib/package-manager.js`

Управлятьобнаружение. 

### ПоддерживатьУправлять

- npm
- pnpm
- yarn
- bun

### обнаружение

1.  `CLAUDE_PACKAGE_MANAGER`
2.  `.claude/package-manager.json`
3. `package.json`  `packageManager` 
4. lock обнаружение
5.  `~/.claude/package-manager.json`
6.  npm

###  API

```javascript
const {
  getPackageManager,           // Управлять
  setPreferredPackageManager,  // 
  setProjectPackageManager,    // 
  getAvailablePackageManagers, // Управлять
  detectFromLockFile,          //  lock обнаружение
  detectFromPackageJson,       //  package.json обнаружение
  getRunCommand,               // 
  getExecCommand,              // 
  getSelectionPrompt,          // Подсказка
  getCommandPattern            // 
} = require('./package-manager');

// Управлять
const { name, config, source } = getPackageManager();
// name: 'pnpm', 'npm', 'yarn',  'bun'
// config: { installCmd, runCmd, execCmd, testCmd, ... }
// source: 'environment', 'project-config', 'lock-file', ...

// 
setPreferredPackageManager('pnpm');

// 
setProjectPackageManager('yarn', '/path/to/project');

// 
const testCmd = getRunCommand('test'); // 'pnpm test'  'npm test' 
const buildCmd = getRunCommand('build');

// 
const eslintCmd = getExecCommand('eslint', '--fix .');
```

### Безопасностьпроверка

`getRunCommand`  `getExecCommand` Проверка:
- , , dash, underscore, dot, slash, @
- , , , dash, dots, slashes, equals, colons, commas, quotes, @

## session-manager.js

****: `scripts/lib/session-manager.js`

Управление сессиями,Предоставлять CRUD . 

### 

- : `~/.claude/session-data/YYYY-MM-DD-[session-id]-session.tmp`
- : `~/.claude/sessions/YYYY-MM-DD-session.tmp`

###  API

```javascript
const {
  parseSessionFilename,      // 
  getSessionPath,            // 
  getSessionContent,         // 
  parseSessionMetadata,      // 
  getSessionStats,           // 
  getSessionTitle,           // 
  getSessionSize,            // ()
  getAllSessions,            // (Поддерживать)
  getSessionById,            //  ID 
  writeSessionContent,       // 
  appendSessionContent,      // 
  deleteSession,             // 
  sessionExists              // проверка
} = require('./session-manager');

// 
const { sessions, total, offset, limit, hasMore } = getAllSessions({
  limit: 50,
  offset: 0,
  date: '2026-05-26',
  search: 'abc123'
});

// 
const session = getSessionById('abc123', true);
// session: { filename, shortId, date, sessionPath, content, metadata, stats, ... }

// 
const metadata = parseSessionMetadata(content);
// metadata: { title, date, started, lastUpdated, project, branch, worktree, completed, inProgress, notes, context }

// 
const stats = getSessionStats(sessionPathOrContent);

//

// 
appendSessionContent(sessionPath, '\n\nMore content...');

// 
deleteSession(sessionPath);
```
