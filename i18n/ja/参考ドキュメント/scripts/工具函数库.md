## utils.js

**路径**: `scripts/lib/utils.js`

跨平台的ツール関数ライブラリ,适用于 Windows、macOS 和 Linux。

### 平台检测

```javascript
const { isWindows, isMacOS, isLinux } = require('./utils');

// isWindows: true on Windows
// isMacOS: true on macOS
// isLinux: true on Linux
```

### ディレクトリ操作

```javascript
const {
  getHomeDir,      // 取得用户主ディレクトリ
  getClaudeDir,    // 取得 ~/.claude ディレクトリ
  getSessionsDir,  // 取得 ~/.claude/session-data ディレクトリ
  getTempDir,      // 取得临时ディレクトリ
  ensureDir,       // 确保ディレクトリ存在,不存在则作成
  findFiles        // 検索匹配パターン的ファイル
} = require('./utils');

// 作成ディレクトリ(如果不存在)
ensureDir('/path/to/dir');

// 検索ファイル
const files = findFiles('/path', '*.js', { recursive: true });
```

### 日期时间

```javascript
const {
  getDateString,      // 戻る YYYY-MM-DD フォーマット
  getTimeString,      // 戻る HH:MM フォーマット
  getDateTimeString   // 戻る YYYY-MM-DD HH:MM:SS フォーマット
} = require('./utils');
```

### セッション和プロジェクト

```javascript
const {
  sanitizeSessionId,    // クリーンアップセッション ID,去除なし效字符
  getSessionIdShort,    // 取得短セッション ID
  getGitRepoName,       // 取得 git 仓库名称
  getProjectName        // 取得プロジェクト名称
} = require('./utils');

// クリーンアップセッション ID
const cleanId = sanitizeSessionId('my-session@123'); // 'my-session-123'

// 取得プロジェクト名
const project = getProjectName();
```

### ファイル操作

```javascript
const {
  readFile,        // セキュリティ読み取りファイル,失敗戻る null
  writeFile,       // 書き込みファイル
  appendFile,      // 追加到ファイル
  replaceInFile,   // 置換ファイル内容
  countInFile,     // 統計パターン出现次数
  grepFile         // 搜索ファイル中的パターン
} = require('./utils');

// 読み取りファイル
const content = readFile('/path/to/file.txt');

// 置換ファイル内容(首次匹配)
replaceInFile('/path', 'old', 'new');

// 置換すべて匹配
replaceInFile('/path', 'old', 'new', { all: true });

// 統計出现次数
const count = countInFile('/path', /pattern/g);

// 搜索并戻る匹配行
const matches = grepFile('/path', /regex/);
// 戻る: [{ lineNumber: 1, content: '...' }]
```

### 字符串处理

```javascript
const { stripAnsi } = require('./utils');

// 移除 ANSI 转义序列(颜色等)
const clean = stripAnsi('\x1b[32mHello\x1b[0m'); // 'Hello'
```

### Hook I/O

```javascript
const { readStdinJson, log, output } = require('./utils');

// 从 stdin 読み取り JSON(用于 hooks)
const input = await readStdinJson({ timeoutMs: 5000 });

// 输出到 stderr(对用户可见)
log('Debug message');

// 输出到 stdout(戻る给 Claude)
output({ result: 'data' });
```

### システムコマンド

```javascript
const { commandExists, runCommand, isGitRepo, getGitModifiedFiles } = require('./utils');

// 検査コマンドは否存在
if (commandExists('git')) { /* ... */ }

// 実行コマンド(仅限白名单コマンド)
const result = runCommand('git rev-parse --show-toplevel');
if (result.success) {
  console.log(result.output);
}

// 検査は否は git 仓库
if (isGitRepo()) { /* ... */ }

// 取得修改的ファイル
const files = getGitModifiedFiles(['\\.js$', '\\.md$']);
```

## package-manager.js

**路径**: `scripts/lib/package-manager.js`

パッケージマネージャー检测和选择ツール。

### サポート的パッケージマネージャー

- npm
- pnpm
- yarn
- bun

### 检测优先级

1. 环境変数 `CLAUDE_PACKAGE_MANAGER`
2. プロジェクト設定 `.claude/package-manager.json`
3. `package.json` 的 `packageManager` 字段
4. lock ファイル检测
5. 全局用户偏好 `~/.claude/package-manager.json`
6. デフォルト值 npm

### コア API

```javascript
const {
  getPackageManager,           // 取得当前プロジェクト的パッケージマネージャー
  setPreferredPackageManager,  // 設定全局偏好
  setProjectPackageManager,    // 設定プロジェクト偏好
  getAvailablePackageManagers, // 取得システム上可用的パッケージマネージャー
  detectFromLockFile,          // 从 lock ファイル检测
  detectFromPackageJson,       // 从 package.json 检测
  getRunCommand,               // 取得実行スクリプト的コマンド
  getExecCommand,              // 取得実行二进制ファイル的コマンド
  getSelectionPrompt,          // 取得交互式选择提示
  getCommandPattern            // 生成コマンド匹配正则
} = require('./package-manager');

// 取得当前パッケージマネージャー
const { name, config, source } = getPackageManager();
// name: 'pnpm', 'npm', 'yarn', 或 'bun'
// config: { installCmd, runCmd, execCmd, testCmd, ... }
// source: 'environment', 'project-config', 'lock-file', ...

// 設定全局偏好
setPreferredPackageManager('pnpm');

// 設定プロジェクト偏好
setProjectPackageManager('yarn', '/path/to/project');

// 取得実行コマンド
const testCmd = getRunCommand('test'); // 'pnpm test' 或 'npm test' 等
const buildCmd = getRunCommand('build');

// 取得実行コマンド
const eslintCmd = getExecCommand('eslint', '--fix .');
```

### セキュリティ検査

`getRunCommand` 和 `getExecCommand` 会検証输入:
- スクリプト名只許可字母、数字、dash、underscore、dot、slash、@
- 参数只許可字母、数字、空白、dash、dots、slashes、equals、colons、commas、quotes、@

## session-manager.js

**路径**: `scripts/lib/session-manager.js`

セッション管理库,提供セッション的 CRUD 操作。

### セッション存储

- 新フォーマット: `~/.claude/session-data/YYYY-MM-DD-[session-id]-session.tmp`
- 旧フォーマット互換性: `~/.claude/sessions/YYYY-MM-DD-session.tmp`

### コア API

```javascript
const {
  parseSessionFilename,      // 解析セッションファイル名
  getSessionPath,            // 取得セッションファイル完整路径
  getSessionContent,         // 読み取りセッション内容
  parseSessionMetadata,      // 解析セッション元データ
  getSessionStats,           // 取得セッション統計
  getSessionTitle,           // 取得セッション标题
  getSessionSize,            // 取得セッションサイズ(人クラス可读フォーマット)
  getAllSessions,            // 取得すべてセッション(サポート分页和筛选)
  getSessionById,            // 按 ID 取得单个セッション
  writeSessionContent,       // 書き込みセッション内容
  appendSessionContent,      // 追加セッション内容
  deleteSession,             // 削除セッション
  sessionExists              // 検査セッションは否存在
} = require('./session-manager');

// 取得すべてセッション
const { sessions, total, offset, limit, hasMore } = getAllSessions({
  limit: 50,
  offset: 0,
  date: '2026-05-26',
  search: 'abc123'
});

// 取得单个セッション
const session = getSessionById('abc123', true);
// session: { filename, shortId, date, sessionPath, content, metadata, stats, ... }

// 解析セッション元データ
const metadata = parseSessionMetadata(content);
// metadata: { title, date, started, lastUpdated, project, branch, worktree, completed, inProgress, notes, context }

// 取得セッション統計
const stats = getSessionStats(sessionPathOrContent);
// stats: { totalItems, completedItems, inProgressItems, lineCount, hasNotes, hasContext }

// 書き込みセッション
writeSessionContent(sessionPath, '# New Session\n\nContent...');

// 追加内容
appendSessionContent(sessionPath, '\n\nMore content...');

// 削除セッション
deleteSession(sessionPath);
```

### セッション元データフォーマット

セッション内容中的元データ:

```markdown
# Session Title

**Date:** 2026-05-26
**Started:** 14:30
**Last Updated:** 15:45
**Project:** my-project
**Branch:** main
**Worktree:** frontend

### Completed
- [x] Task 1
- [x] Task 2

### In Progress
- [ ] Task 3

### Notes for Next Session
Remember to check the config file.

### Context to Load
```
previous-context-content
```
```

const {
